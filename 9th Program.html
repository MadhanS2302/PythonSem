Program A: Area Calculation Using Modules

        
# module: shape.py
from math import sqrt

def area_square(a):
    return a * a

def area_circle(r):
    return 3.14 * r * r

def area_rectangle(l, b):
    return l * b

def area_triangle(b, h):
    return 0.5 * b * h

def area_hexagon(s):
    return (3 * sqrt(3) / 2) * s * s

def area_cylinder(h, r):
    return 2 * 3.14 * r * (h + r)

# program: main.py
import shape

print("Different shapes:")
print("\n 1. Square \n 2. Circle \n 3. Rectangle \n 4. Triangle \n 5. Hexagon \n 6. Cylinder \n 7. Exit \n")

while True:
    ch = int(input("Enter your choice: "))
    if ch == 1:
        a = float(input("Enter the side of the square: "))
        print("Area of square =", shape.area_square(a))
    elif ch == 2:
        r = float(input("Enter the radius of the circle: "))
        print("Area of circle =", shape.area_circle(r))
    elif ch == 3:
        l = float(input("Enter the length of the rectangle: "))
        b = float(input("Enter the breadth of the rectangle: "))
        print("Area of rectangle =", shape.area_rectangle(l, b))
    elif ch == 4:
        b = float(input("Enter the base of the triangle: "))
        h = float(input("Enter the height of the triangle: "))
        print("Area of triangle =", shape.area_triangle(b, h))
    elif ch == 5:
        s = float(input("Enter the side of the hexagon: "))
        print("Area of hexagon =", shape.area_hexagon(s))
    elif ch == 6:
        h = float(input("Enter the height of the cylinder: "))
        r = float(input("Enter the radius of the cylinder: "))
        print("Area of cylinder =", shape.area_cylinder(h, r))
    elif ch == 7:
        print("Exiting...")
        break
    else:
        print("Invalid choice, please try again.")

Program B: Flower Drawing Using Turtle Graphics

        
from turtle import Turtle, Screen

def draw_petal(turtle, radius):
    heading = turtle.heading()
    turtle.circle(radius, 60)
    turtle.left(120)
    turtle.circle(radius, 60)
    turtle.setheading(heading)

my_radius = int(input("Radius of flower: "))
my_petals = int(input("How many petals: "))

bob = Turtle()
for _ in range(my_petals):
    draw_petal(bob, my_radius)
    bob.left(360 / my_petals)

bob.hideturtle()

screen = Screen()
screen.exitonclick()
