Program A: FizzBuzz

        
def fizzbuzz():
    num = int(input("Enter the number"))
    for i in range(1, num + 1):
        if i % 3 == 0 and i % 5 == 0:
            print("Fizzbuzz")
        elif i % 3 == 0:
            print("Fizz")
        elif i % 5 == 0:
            print("Buzz")
        else:
            print(i)

fizzbuzz()

Program B: Password Validation

        
import re
p = input("ENTER YOUR PASSWORD:")
x = True
while x:
    if len(p) < 6 or len(p) > 12:
        break
    elif not re.search("[a-z]", p):
        break
    elif not re.search("[0-9]", p):
        break
    elif not re.search("[A-Z]", p):
        break
    elif not re.search("[$#@]", p):
        break
    elif re.search("/s", p):
        break
    else:
        print("Valid password")
        x = False
        break
if x:
    print("Not a valid password")
