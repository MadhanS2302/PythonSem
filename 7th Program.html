Program A: Method Overloading for Area Calculation

        
class compute:
    def area(self, x=None, y=None, z=None):
        if x != None and y != None and z != None:
            return x + y + z
        elif x != None and y != None:
            return x * y
        elif x != None:
            return 3.14 * x * x
        else:
            return 0

obj = compute()
r = int(input("Enter radius value: "))
print("Area of circle =:", obj.area(r))

l = int(input("Enter length of rectangle: "))
b = int(input("Enter breadth of rectangle: "))
print("Area of rectangle:", obj.area(l, b))

a = int(input("Enter a side value: "))
b = int(input("Enter base value: "))
c = int(input("Enter another side value: "))
print("Perimeter of triangle is:", obj.area(a, b, c))

Program B: Vehicle Rental System with Polymorphism

        
class vehicle:
    def __init__(self, base_rate):
        self.base_rate = base_rate
    
    def rent(self, duration, customer_type):
        raise NotImplementedError("Subclass must implement this method")

class Car(vehicle):
    def __init__(self):
        super().__init__(base_rate=20)

    def rent(self, duration, customer_type):
        discount = 0.1 if customer_type == "Premium" else 0
        cost = duration * self.base_rate * (1 - discount)
        return f"Car rented for {duration} hours. Cost: {cost:.2f} USD"

class Bike(vehicle):
    def __init__(self):
        super().__init__(base_rate=5)

    def rent(self, duration, customer_type):
        discount = 0.05 if customer_type == "Premium" else 0
        cost = duration * self.base_rate * (1 - discount)
        return f"Bike rented for {duration} hours. Cost: {cost:.2f} USD"

class Truck(vehicle):
    def __init__(self):
        super().__init__(base_rate=50)

    def rent(self, duration, customer_type):
        discount = 0.15 if customer_type == "Premium" else 0
        cost = duration * self.base_rate * (1 - discount)
        return f"Truck rented for {duration} hours. Cost: {cost:.2f} USD"

vehicles = [Car(), Bike(), Truck()]
customer_type = "Premium"

for vehicle in vehicles:
    print(vehicle.rent(4, customer_type))
