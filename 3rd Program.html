Program A: Guess the Number

        
guessed_number = 45

print("Find the guessed number")
x = int(input("Enter your choice: "))

if x == guessed_number:
    print("Congratulations, you've found it!")
else:
    while x != guessed_number:
        if x > guessed_number:
            print("Too high from target")
        elif x < guessed_number:
            print("Too low from target")
        x = int(input("Enter your choice: "))
    print("Congratulations, you've found it!")

Program B: Factorial Calculation

        
def factorial(n):
    result = 1
    for i in range(1, n + 1):
        result *= i  
    return result

number = int(input("Enter the number: "))
fact = factorial(number)
print(f"The factorial of {number} is {fact}")
