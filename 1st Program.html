Python Program: Contact Book Example

        
contactbook = {}

def addcontact(name, mobile, email, address):
    address = tuple(address)
    email = tuple(email)
    contactbook[name] = {
        'name': name,
        'mobile': mobile,
        'email': email,
        'address': address,
    }

def viewcontact():
    print("contactbook")
    for x, y in contactbook.items():
        print(x, y)

addcontact("Madhan", '9789609971', ['Madhan@gmail.com', 'Madhan007@gmail.com'], ["Saravanampatti"])
viewcontact()
